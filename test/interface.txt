            function f(n, m, a[][]);
            function g(i, j);

            main {
                read n, m;
                for i to n {
                    for j to m {
                        read a[i][j];
                    }
                }

                call r = f(n, m, a);
                write r;

                for i to n {
                    for j to m {
                        call s = g(i, j);
                        write s;
                    }
                }
            }
