interface exampleinterface {
    global int N, M;
    global int A[];
    global int S;

    function int solve(int u);

    main {
        input N, M;
        for(i : 1..N) {
            input A[i];
        }
        local int u;
        input u;
        call S = solve(u);
        output S;
    }
}
