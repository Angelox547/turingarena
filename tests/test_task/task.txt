interface exampleinterface {
    global int N, M;
    global int64[] A;
    global int64 S;

    function int solve(int u);

    main {
        input N, M;
        alloc A : 1..N;
        for(i : 1..N) {
            input A[i];
        }
        local int u;
        input u;
        call solve(u) -> S;
        output S;
    }
}
