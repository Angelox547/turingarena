interface alien {
   int find_centre(int N, int Xi,int Yi);
   callback int examine(int x, int y);
   callback int solution(int x, int y);

   globals {
      int N;
      int M;
      int A[N];
      int B[N][M];
   }

   setters {
      first {
          N, M;
      }
   }

   getters {
      result {
          A, B
      }
   }

   stages {
      func main {
        set{first}
        find_centre()
        get{result}
      }

      func find_centre {
          loop {
            examine()
          }
          loop(1, 10000, i) {
              examine(i)
              switch {
                  examine()
                  solution()
              }
          } 
          solution
      }

      func solution {
          times(1, N) {
              examine()
          }
      }

   }

}

driver main {
  int evaluate();
}

scenario test1 {
  phase evaluate {
    slot solution interface alien;
    driver main evaluate();
  }
}
