var int n;
var int[] a;

function compute();
function length() -> int;
function element(int i) -> int;

init {
    read n;
    alloc a : n;
    for(i : n) {
        read a[i];
    }
}

main {
    call compute();

    var int l;
    call length() -> l;
    write l;
    for(i : l) {
        var int x;
        call element(i) -> x;
        write x;
    }
}
