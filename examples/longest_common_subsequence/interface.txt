// compute the longest common subsequence, return the sequence length
int compute(int M, int X[], int N, int Y[]);

// return the i-th element of the sequence
int element(int i);

main {
    read M;
    for i to M {
        read X[i];
    }

    read N;
    for i to N {
        read Y[i];
    }

    call l = compute(M, X, N, Y);

    write l;

    for i to l {
        call x = element(i);
        write x;
    }
}
