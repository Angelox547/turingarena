void init(int N, int Q, int D[], int adj[][]);

int is_there_a_path(int u, int v);

main {
    read N, Q;

    for u to N {
        read D[u];
        for i to D[u] {
            read adj[u][i];
        }
    }

    call init(N, Q, D, adj);

    for q to Q {
        read u, v;
        call ans = is_there_a_path(u, v);
        write ans;
    }
}
